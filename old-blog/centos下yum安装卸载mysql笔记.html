# centos下yum安装卸载mysql笔记

发表于 2013-11-09 |

# 检查老版本mysql

查看CentOS自带mysql是否已安装。 输入：

yum list installed  grep mysql
1

# 卸载CentOS系统自带mysql

执行下面语句

yum remove mysql mysql-server mysql-libs compat-mysql51
rm -rf varlibmysql
rm etcmy.cnf
1
2
3

查看是否还有mysql：

rpm -qagrep mysql
1

如果存在，继续删除。

然后

yum update
1

# 查看yum库上的mysql版本信息

yum -y list mysql
1

# 使用yum安装mysql

yum -y install mysql-server mysql mysql-devel
1

查看安装mysql数据库版本

rpm -qi mysql-server
1

# 开机自动启动

chkconfig mysqld on
1

# 启动mysql

service mysql start
1

# 为root用户设置密码为root

mysqladmin -u root password 'root'
1

# 为root用户设置远程访问

grant all privileges on . to 'root'@'%' identified by 'root' with grant option;
1

# 刷新权限

flush privileges;
1

# 添加防火墙过滤

iptables -I INPUT -p tcp --dport 3306 -j ACCEPT
etcrc.dinit.diptables save
service iptables  restart
